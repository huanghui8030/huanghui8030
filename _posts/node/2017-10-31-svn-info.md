通过nodeJS获取svn记录信息，用到svn-info插件，

用法如下：

```js
var svn = require('svn-info');

svn('svnUrl', 'HEAD', function(err, info) {//其中svnUrl为url地址
  if(err) {
    throw err;
  }
  console.log(info.lastChangedAuthor);
  console.log(info.lastChangedDate)
  console.log(info);
});

```
